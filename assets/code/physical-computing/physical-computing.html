#include FastLED.h

#define LED_PIN 3
#define NUM_LEDS 30
#define BRIGHTNESS 255
#define LED_TYPE WS2811
#define COLOR_ORDER RGB
CRGB leds[NUM_LEDS];

#define OUT_PIN_1 2
#define OUT_PIN_2 4
#define OUT_PIN_3 7

uint8_t output_value_1 = 0;
uint8_t output_value_2 = 0;
uint8_t output_value_3 = 0;

void setup() {
FastLED.addLeds(leds, NUM_LEDS).setCorrection(TypicalLEDStrip);
FastLED.setBrightness(BRIGHTNESS);

pinMode(OUT_PIN_1, INPUT);
pinMode(OUT_PIN_2, INPUT);
pinMode(OUT_PIN_3, INPUT);
}

void loop() {
// Sensor 1
output_value_1 = digitalRead(OUT_PIN_1);
if (output_value_1) {
mySubsection1(CRGB(255, 0, 0)); // Set subsection to red
} else {
mySubsection1(CRGB(0, 0, 0)); // Turn off subsection
}

// Sensor 2
output_value_2 = digitalRead(OUT_PIN_2);
if (output_value_2) {
mySubsection2(CRGB(255, 0, 0)); // Set subsection to red
} else {
mySubsection2(CRGB(0, 0, 0)); // Turn off subsection
}

// Sensor 3
output_value_3 = digitalRead(OUT_PIN_3);
if (output_value_3) {
mySubsection3(CRGB(255, 0, 0)); // Set subsection to red
} else {
mySubsection3(CRGB(0, 0, 0)); // Turn off subsection
}

FastLED.show();
}

// Functions to control LED subsections
void mySubsection1(CRGB color) {
leds[0] = color;
leds[1] = color;
leds[2] = color;
leds[3] = color;
leds[4] = color;
leds[5] = color;
}

void mySubsection2(CRGB color) {
leds[6] = color;
leds[7] = color;
leds[8] = color;
leds[9] = color;
leds[10] = color;
leds[11] = color;
}

void mySubsection3(CRGB color) {
leds[12] = color;
leds[13] = color;
leds[14] = color;
leds[15] = color;
leds[16] = color;
leds[17] = color;
}
